        anchorIndex[_hyperRoot] = idx;

    function validateAnchor(

        bytes32 _approvalHash

        uint256 idx = anchorIndex[_hyperRoot];
        require(idx < anchors.length, "Anchor not found");

        anchors[idx].isValid = true;

        emit AnchorValidated(_hyperRoot, true, "Multi-sig validated");

     * @notice Verifica se um hiper-root estГЎ ancorado e vГЎlido

    function verifyAnchor(bytes32 _hyperRoot) external view returns (bool, uint256) {

        if (idx >= anchors.length) return (false, 0);
        return (anchors[idx].isValid, idx);

    // FUNГҮГ•ES DE GOVERNANГҮA

    function addParticipant(
        address _addr,
        ParticipantType _pType,
        string calldata _institution,
        bool _isValidator,
        bytes32 _approvalHash
    ) external multiSigApproved(_approvalHash) {
        participants[_addr] = Participant({
            addr: _addr,
            pType: _pType,
            institution: _institution,
            isValidator: _isValidator,
            isActive: true
        });
        participantList.push(_addr);

        emit GovernanceAction("ADD_PARTICIPANT", _addr, block.timestamp);
    }

    /**
     * @notice Veto de Гўncora (apenas TCU)
     * @param _hyperRoot Hiper-root a vetar

    function vetoAnchor(bytes32 _hyperRoot) external onlyPatron {

        // TCU specific check
        require(
            keccak256(bytes(participants[msg.sender].institution)) == keccak256(bytes("TCU")),
            "Only TCU can veto"
        );

        anchors[idx].isValid = false;

        emit AnchorValidated(_hyperRoot, false, "TCU veto applied");

    }

    // в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ

    // в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ

class ZKBridge:
    """Ponte ZK entre RBB e Catedral."""

    def __init__(self):
        self.circuit_hash = self._compile_circuit()
        self.proving_key = None
        self.verification_key = None

    def _compile_circuit(self) -> str:
        """Compila circuito Circom rbb_verify.circom."""
        circuit = """
        template RBBVerify(blocks, txPerBlock) {
            signal input blockHeaders[blocks];
            signal input merkleProofs[blocks][txPerBlock];
            signal input stateRoot;
            signal output valid;

            // Verify chain of block hashes
            // Verify state transitions
            // Verify Merkle inclusion of anchor tx

            valid <== 1;  // Simplified
        }
        """
        return hashlib.sha3_256(circuit.encode()).hexdigest()[:16]

    def generate_proof(self, block: RBBBlock, anchor: CathedralAnchor) -> bytes:
        """Gera ZK proof de inclusГЈo do anchor na RBB."""
        # Em produГ§ГЈo: snarkjs groth16 fullprove
        witness = {
            "blockHash": block.hash,
            "stateRoot": block.state_root,
            "anchorHyperRoot": anchor.hyper_root,
            "temporalSeal": anchor.temporal_seal
        }
        proof = hashlib.sha3_256(json.dumps(witness, sort_keys=True).encode()).digest()
        print(f"[ZK] Proof generated: {proof.hex()[:32]}...")
        return proof

    def verify_proof(self, proof: bytes, public_inputs: dict) -> bool:
        """Verifica ZK proof no lado da Catedral."""
        # Em produГ§ГЈo: snarkjs groth16 verify
        expected = hashlib.sha3_256(json.dumps(public_inputs, sort_keys=True).encode()).digest()
        is_valid = proof == expected
